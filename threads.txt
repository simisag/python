#Q1.Create a threading process such that it sleeps for 5 seconds and then prints out a message.

import time
import threading

def p():
   time.sleep(5)
   print("Morning")

threading.Thread(target=p).start()



#Q2. Make a thread that prints numbers from 1-10, waits for 1 sec between 

import time
import threading

def p():
   for i in range(1,11):
      print(i)
      time.sleep(1)

threading.Thread(target=p).start()
